# ポインタと配列の関係
- ある要素へのポインタに整数nを足すと、n個後ろの要素へのポインタが得られる
- `&a[n]`と`a+n`の値は等しくメモリを表す
- `a[n]`と`*(a+n)`の値は等しくメモリの中身を表す
  
<br>

### ソースコード

```c
#include<stdio.h>

void main(void) {
	int a[5] = { 31,32,34,35,36 };
	double b[5] = { 31.51,32.52,34.44,35.84,36.23 };

	printf("-------------------------------------int型の例-------------------------------------\n");
	printf("【配列の要素数の取り方】\n");
	printf("sizeof(a) = %zu：sizeof(a[0]) = %zu\n", sizeof(a),sizeof(a[0]));			//20バイト （int型4バイト * 要素数5）
	printf("sizoof(a) / sizeof(a[0]) = %d\n", sizeof(a)/sizeof(a[0]));	//4バイト （int型4バイト）

	printf("\n");

	printf("【&a[n]とa+nの値は等しい】\n");
	printf("&a[0] = %p\n", &a[0]);					//配列aの先頭メモリ
	printf("a = %p\n", a);							//配列aの先頭メモリ
	printf("&a[1] = %p\n", &a[1]);					//配列aの先頭メモリ
	printf("a+1 = %p\n", a + 1);					//配列aの先頭メモリ + 4バイト
	printf("&a[2] = %p\n", &a[2]);					//配列aの先頭メモリ
	printf("a+2 = %p\n", a + 2);					//配列aの先頭メモリ + 2バイト

	printf("\n");

	printf("【a[n]と*(a+n)は等価】\n");
	printf("a[0] = %d\n", a[0]);					//31 (a[0] = 31)
	printf("*a = %d\n", *a);						//31（配列aのポインタ）
	printf("a[1] = %d\n", a[1]);					//32（a[1] = 32)
	printf("*(a+1) = %d\n", *(a + 1));				//32（31+4）
	printf("a[2] = %d\n", a[2]);					//34（a[2])
	printf("*(a+2) = %d\n", *(a + 2));				//34（31+8）

	printf("\n");

	printf("-------------------------------------doule型の例-------------------------------------\n");
	printf("【配列の要素数の取り方】\n");
	printf("sizeof(b) = %zu：sizeof(b[0]) = %zu\n", sizeof(b), sizeof(b[0]));			//40バイト （double型8バイト * 要素数5）
	printf("sizeof(b) / sizeof(b[0]) = %d\n", sizeof(b) / sizeof(b[0]));	//8バイト （double型8バイト）

	printf("\n");

	printf("【&b[n]とb+nの値は等しい】\n");
	printf("&b[0] = %p\n", &b[0]);					//配列bの先頭メモリ
	printf("b = %p\n", b);							//配列bの先頭メモリ
	printf("&b[1] = %p\n", &b[1]);					//配列b[1]の先頭メモリ
	printf("b+1 = %p\n", b + 1);					//配列aの先頭メモリ + 8バイト
	printf("&b[2] = %p\n", &b[2]);					//配列b[2]の先頭メモリ
	printf("b+2 = %p\n", b + 2);					//配列bの先頭メモリ +8

	printf("\n");

	printf("【b[n]と*(b+n)は等価】\n");
	printf("b[0] = %f\n", b[0]);					//31.51 (b[0] = 31.51)
	printf("*b = %f\n", *b);						//31.51（配列bのポインタ）
	printf("b[1] = %f\n", b[1]);					//32.52（b[1] = 32.52)
	printf("*(b+1) = %f\n", *(b + 1));				//32.52（31+4）
	printf("b[2] = %f\n", b[2]);					//34.34（a[2])
	printf("*(b+2) = %f\n", *(b + 2));				//34.44（31+8）

	return 0;
}
```
